<!-- Admin: neuen Spielzug hinzufügen -->

<div id="admin-panel">

    <h2>Admin: Spielzug hinzufügen</h2>

    <input type="text" id="play-input" placeholder="Spielzug">

    <button onclick="addPlay()">Speichern</button>

</div>


<ul id="play-list"></ul>


<script>

const password = 'coach2025!'; // Admin-Passwort


async function loadPlays() {

    const res = await fetch('api/plays.json');

    const plays = await res.json();

    const list = document.getElementById('play-list');

    list.innerHTML = '';

    plays.forEach(p => {

        const li = document.createElement('li');

        li.textContent = p;

        list.appendChild(li);

    });

}


async function addPlay() {

    const play = document.getElementById('play-input').value;

    if (!play) return alert('Bitte Spielzug eingeben');


    const formData = new FormData();

    formData.append('password', password);

    formData.append('play', play);


    const res = await fetch('api/save_play.php', {

        method: 'POST',

        body: formData

    });


    const data = await res.json();

    if (data.success) {

        document.getElementById('play-input').value = '';

        loadPlays();

    } else {

        alert(data.error || 'Fehler beim Speichern');

    }

}


// Initial laden

loadPlays();

</script>
